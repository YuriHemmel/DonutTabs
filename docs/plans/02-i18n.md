# DonutTabs — Plano 2: Internacionalização (i18n) + `AppError` estruturado

> **Para agentes executores:** SUB-SKILL OBRIGATÓRIA: Use `superpowers:subagent-driven-development` (recomendada) ou `superpowers:executing-plans` para implementar este plano tarefa-a-tarefa. Passos usam checkbox (`- [ ]`) para rastreamento.

**Meta:** Tornar a aplicação traduzível antes de crescer a UI na fase de Settings. Introduzir `react-i18next` no frontend com arquivos de locale (`pt-BR` e `en`), ativar `appearance.language` no schema, e evoluir `AppError` no Rust de strings livres em português para **códigos estruturados + contexto** (Seção 6.7 do [Plano.md](../Plano.md)). Toda mensagem voltada ao usuário passa a sair de uma chave de tradução; logs internos continuam em inglês técnico.

**Arquitetura:** Nenhuma mudança estrutural. No frontend, adiciona-se `src/core/i18n.ts` + `src/locales/{pt-BR,en}.json` + um mapper `src/core/errors.ts` que converte códigos do Rust em strings traduzidas. No Rust, `AppError` ganha variantes com `{ code, context }` em vez de mensagens prontas — formato preservado via `#[serde(tag = "kind", content = "message")]` para o frontend receber discriminado.

**Stack adicional:** `react-i18next`, `i18next`, `i18next-browser-languagedetector` (frontend). Nenhuma crate nova no Rust — apenas refactor.

**Fora desta slice (vem no Plano 3+):**
- Janela de Settings (que consumirá este i18n).
- Fatia "+", CRUD de abas, ShortcutRecorder.
- Perfis, menu de contexto, favicons, drag-and-drop.

- [ ] **Step 4.2 — Criar `src/core/i18n.ts`**

```ts
import i18next, { type i18n as I18nInstance } from "i18next";
import { initReactI18next } from "react-i18next";
import ptBr from "../locales/pt-BR.json";
import en from "../locales/en.json";
import type { Language } from "./types/Language";

export type ResolvedLocale = "pt-BR" | "en";

/**
 * Resolve o locale efetivo a partir da preferência do config e do idioma do navegador.
 * - `ptBr` / `en` → uso direto.
 * - `auto` → navigator.language começando com "pt" → pt-BR; caso contrário en.
 */
export function resolveLanguage(
  configLanguage: Language,
  navigatorLanguage: string | undefined
): ResolvedLocale {
  if (configLanguage === "ptBr") return "pt-BR";
  if (configLanguage === "en") return "en";
  // auto
  if (navigatorLanguage && navigatorLanguage.toLowerCase().startsWith("pt")) {
    return "pt-BR";
  }
  return "en";
}

/**
 * Cria uma instância isolada do i18next. Usada em testes;
 * em produção, `initI18n` abaixo configura a instância global.
 */
export async function createI18n(locale: ResolvedLocale): Promise<I18nInstance> {
  const instance = i18next.createInstance();
  await instance.init({
    resources: {
      "pt-BR": { translation: ptBr },
      en: { translation: en },
    },
    lng: locale,
    fallbackLng: "en",
    interpolation: { escapeValue: false },
  });
  return instance;
}

/**
 * Inicializa a instância global do i18next (a que `useTranslation` consome).
 * Deve ser chamada uma vez no entrypoint de cada janela, antes do render.
 */
export async function initI18n(configLanguage: Language): Promise<ResolvedLocale> {
  const locale = resolveLanguage(
    configLanguage,
    typeof navigator !== "undefined" ? navigator.language : undefined
  );
  await i18next
    .use(initReactI18next)
    .init({
      resources: {
        "pt-BR": { translation: ptBr },
        en: { translation: en },
      },
      lng: locale,
      fallbackLng: "en",
      interpolation: { escapeValue: false },
    });
  return locale;
}

/**
 * Permite trocar o idioma em runtime (ex: usuário alterou no Settings — Plano 3).
 * No Plano 2 ainda não há UI para isso; a função fica pronta.
 */
export async function changeLanguage(configLanguage: Language): Promise<ResolvedLocale> {
  const locale = resolveLanguage(
    configLanguage,
    typeof navigator !== "undefined" ? navigator.language : undefined
  );
  await i18next.changeLanguage(locale);
  return locale;
}
```

```rust
use serde::Serialize;
use std::collections::BTreeMap;
use thiserror::Error;

pub type ErrorContext = BTreeMap<String, String>;

#[derive(Debug, Clone, Error, Serialize, PartialEq, Eq)]
#[serde(tag = "kind", content = "message", rename_all = "camelCase")]
pub enum AppError {
    #[error("config error: {code}")]
    Config {
        code: String,
        #[serde(default)]
        context: ErrorContext,
    },
    #[error("shortcut error: {code}")]
    Shortcut {
        code: String,
        #[serde(default)]
        context: ErrorContext,
    },
    #[error("launcher error: {code}")]
    Launcher {
        code: String,
        #[serde(default)]
        context: ErrorContext,
    },
    #[error("window error: {code}")]
    Window {
        code: String,
        #[serde(default)]
        context: ErrorContext,
    },
    #[error("io error: {code}")]
    Io {
        code: String,
        #[serde(default)]
        context: ErrorContext,
    },
}

impl AppError {
    /// Helper: constrói um erro com contexto a partir de pares chave/valor.
    pub fn config(code: impl Into<String>, ctx: &[(&str, String)]) -> Self {
        AppError::Config {
            code: code.into(),
            context: ctx.iter().map(|(k, v)| (k.to_string(), v.clone())).collect(),
        }
    }

    pub fn launcher(code: impl Into<String>, ctx: &[(&str, String)]) -> Self {
        AppError::Launcher {
            code: code.into(),
            context: ctx.iter().map(|(k, v)| (k.to_string(), v.clone())).collect(),
        }
    }

    pub fn window(code: impl Into<String>, ctx: &[(&str, String)]) -> Self {
        AppError::Window {
            code: code.into(),
            context: ctx.iter().map(|(k, v)| (k.to_string(), v.clone())).collect(),
        }
    }
}

impl From<std::io::Error> for AppError {
    fn from(e: std::io::Error) -> Self {
        AppError::Io {
            code: "io_generic".into(),
            context: [("reason".to_string(), e.to_string())]
                .into_iter()
                .collect(),
        }
    }
}

impl From<serde_json::Error> for AppError {
    fn from(e: serde_json::Error) -> Self {
        AppError::Config {
            code: "json_parse".into(),
            context: [("reason".to_string(), e.to_string())]
                .into_iter()
                .collect(),
        }
    }
}

pub type AppResult<T> = Result<T, AppError>;
```

- [ ] **Step 8.2 — Criar `src/core/errors.ts`**

```ts
import type { TFunction } from "i18next";

export interface AppError {
  kind: "config" | "shortcut" | "launcher" | "window" | "io";
  message: {
    code: string;
    context: Record<string, string>;
  };
}

export function isAppError(value: unknown): value is AppError {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  if (typeof v.kind !== "string") return false;
  if (typeof v.message !== "object" || v.message === null) return false;
  const m = v.message as Record<string, unknown>;
  return typeof m.code === "string";
}

/**
 * Converte `code` (snake_case) em `camelCase` para casar com chaves do JSON.
 * Ex: "items_per_page_out_of_range" → "itemsPerPageOutOfRange".
 */
function snakeToCamel(s: string): string {
  return s.replace(/_([a-z])/g, (_, c) => c.toUpperCase());
}

/**
 * Produz mensagem traduzida. Ordem de resolução:
 * 1. `errors.{kind}.{camelCode}` se existir.
 * 2. `errors.{kind}.unknown` com `{ code }` no contexto.
 * 3. `errors.fallback` com `{ kind }` como último recurso.
 */
export function translateAppError(err: unknown, t: TFunction): string {
  if (!isAppError(err)) {
    return t("errors.fallback", { kind: typeof err });
  }
  const camel = snakeToCamel(err.message.code);
  const specificKey = `errors.${err.kind}.${camel}`;
  const translated = t(specificKey, { ...err.message.context, defaultValue: "" });
  if (translated) return translated;
  // Fallback 1: errors.{kind}.unknown
  const unknownKey = `errors.${err.kind}.unknown`;
  const unknownTranslated = t(unknownKey, { code: err.message.code, defaultValue: "" });
  if (unknownTranslated) return unknownTranslated;
  // Fallback 2: errors.fallback
  return t("errors.fallback", { kind: err.kind });
}
```
